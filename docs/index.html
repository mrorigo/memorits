import { Memori } from 'memorits';

const memori = new Memori({ databaseUrl: 'sqlite:./memories.db' });
await memori.enable();

  model: 'gpt-4o-mini',
  importance: 'high',
  category: 'technical-decision'
});

// Search with advanced filtering
const memories = await memori.searchMemories('urgent issues', {
  limit: 20,
  minImportance: 'high',
  categories: ['essential', 'contextual'],
  temporalFilters: { relativeExpressions: ['last week'] }
});
